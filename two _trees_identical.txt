int identicalTrees(struct node* a, struct node* b)
{
    /*1. both empty */
    if (a==NULL && b==NULL)
        return 1;
 
    /* 2. both non-empty -> compare them */
    if (a!=NULL && b!=NULL)
    {
        
     if(a->data == b->data &&
        identicalTrees(a->left, b->left) &&
        identicalTrees(a->right, b->right))
     return 1;    
}
     
    /* 3. one empty, one not -> false */
    return 0;
}
